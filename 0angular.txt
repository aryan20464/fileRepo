apiVersion: kafka.strimzi.io/v1beta2
kind: KafkaUser
metadata:
  name: my-user
  namespace: rforex-kafka
  labels:
    strimzi.io/cluster: my-cluster
spec:
  authentication:
    type: scram-sha-512

  authorization:
    type: simple
    acls:

      # REQUIRED for metadata
      - resource:
          type: cluster
        operations:
          - Describe
          - DescribeConfigs

      # REQUIRED to list topics
      - resource:
          type: topic
          name: "*"
          patternType: literal
        operations:
          - Describe
          - Read
          - Write
          - Create

      # REQUIRED for consumers (even if unused)
      - resource:
          type: group
          name: "*"
          patternType: literal
        operations:
          - Describe
          - Read
