apiVersion: kafka.strimzi.io/v1beta2
kind: KafkaUser
metadata:
  name: my-user
  namespace: rforex-kafka
  labels:
    strimzi.io/cluster: my-cluster
spec:
  authentication:
    type: scram-sha-512
  authorization:
    type: simple
    acls:
      - resource:
          type: topic
          name: '*'
          patternType: literal
        operation: Read
      - resource:
          type: topic
          name: '*'
          patternType: literal
        operation: Write
      - resource:
          type: topic
          name: '*'
          patternType: literal
        operation: Create
      - resource:
          type: group
          name: '*'
          patternType: literal
        operation: Read





apiVersion: kafka.strimzi.io/v1beta2
kind: KafkaBridge
metadata:
  name: my-bridge
  namespace: rforex-kafka
spec:
  replicas: 1

  bootstrapServers: my-cluster-kafka-bootstrap:9093

  authentication:
    type: scram-sha-512
    username: my-user
    passwordSecret:
      secretName: my-user
      password: password

  tls:
    trustedCertificates:
      - secretName: my-cluster-cluster-ca-cert
        certificate: ca.crt

  http:
    port: 8080

  enableMetrics: false
