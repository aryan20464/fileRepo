---
# ----------------------------------------
# Deployment: App + Alloy Sidecar
# ----------------------------------------
apiVersion: apps/v1
kind: Deployment
metadata:
  name: dbcon-app
  namespace: uatcmm
spec:
  replicas: 1
  selector:
    matchLabels:
      app: dbcon-app
  template:
    metadata:
      labels:
        app: dbcon-app
    spec:
      containers:
        - name: dbcon-app
          image: tkgsharedharbor.corp.ad.sbi/uat-cmm/dbcon:3.0
          ports:
            - containerPort: 4320
          securityContext:
            allowPrivilegeEscalation: false
            capabilities:
              drop:
                - ALL
            runAsNonRoot: true
            runAsUser: 1001
            seccompProfile:
              type: RuntimeDefault
          volumeMounts:
            - name: app-logs
              mountPath: /app/logs
          env:
            - name: LOG_DIR
              value: /app/logs

        - name: alloy
          image: tkgsharedharbor.corp.ad.sbi/itss/grafana-alloy:1.8.3
          args: ["run", "--storage.path=/var/lib/alloy", "/etc/alloy/config.alloy"]
          volumeMounts:
            - name: alloy-config-sidecar
              mountPath: /etc/alloy
              readOnly: true
            - name: app-logs
              mountPath: /logs
            - name: alloy-data
              mountPath: /var/lib/alloy
          securityContext:
            runAsNonRoot: true
            runAsUser: 1000
            allowPrivilegeEscalation: false
            capabilities:
              drop: ["ALL"]

      volumes:
        - name: app-logs
          emptyDir: {}
        - name: alloy-config-sidecar
          configMap:
            name: alloy-config-sidecar
        - name: alloy-data
          emptyDir: {}
