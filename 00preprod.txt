  customFilter(value: string, field: string, mode: string): void {
    const filteredValue = value.toLowerCase();
    this.filteredcircleUserAndEmailsArray = this.circleUserAndEmailsArray.filter((sdata:any) => {
      switch (field) {
        case 'vendorRes':
          return this.vendorArray[sdata.vendorRes].toLowerCase().includes(filteredValue);
        default:
          return sdata[field].toString().toLowerCase().includes(filteredValue);
      }
    });
  }

  customFilterVendor(value: string, field: string, mode: string): void {
    const filteredValue = value.toLowerCase();
    this.filteredvendorEmailsArray = this.vendorEmailsArray.filter((sdata:any) => {
      switch (field) {
        case 'vendorRes':
          return this.vendorArray[sdata.vendorRes].toLowerCase().includes(filteredValue);
        default:
          return sdata[field].toString().toLowerCase().includes(filteredValue);
      }
    });
  }


  // Filter for branch report at GMUK for daily indents. 22-06-2025
  customFilterReportDate(value: string, field: string, mode: string): void {
    const filteredValue = value.toLowerCase();
    this.filteredbranchReportData = this.branchReportData.filter(sdata => {
      switch (field) {
        case 'createdAt':
          return new Date(sdata['createdAt']).toISOString().substring(0,10).includes(value);
        case 'requestedDate':
          return new Date(sdata['requestedDate']).toISOString().substring(0,10).includes(value);
        case 'receivedDate':
          return new Date(sdata['receivedDate']).toISOString().substring(0,10).includes(value);
        case 'indentStatus':
          return this.appStatus[sdata[field]].toString().toLowerCase().includes(filteredValue);
        case 'currencyType':
          return (this.currencyData[sdata[field]]+sdata[field]).toString().toLowerCase().includes(filteredValue);
        case 'branchName':
          return (sdata['branchName']+this.circleNames[sdata['circleCode']]).toString().toLowerCase().includes(filteredValue);
        case 'circleName':
          return (this.circleNames[sdata['circleCode']]).toString().toLowerCase().includes(filteredValue);
        default:
          return sdata[field].toString().toLowerCase().includes(filteredValue);
      }
    });
  }

  onBranchSelect(event: any) {
    //console.log('Selected branch:', event.value);
    const bcode = event.value;
    const apiUrl = `${environment.rfmIndentService}/getBranchVendorMapping?bcode=`+bcode; // Replace with your API endpoint
    let mailEntries = [];
    this.http.get<any[]>(apiUrl).subscribe(
      (data:any) => {
        this.branchVendorList = data; // Store response data in records variable
        //console.log(this.branchVendorList);
        //console.log("length"+this.branchVendorList.length)
        mailEntries = this.branchVendorList["vendorEmail"].split(";");
        //console.log(mailEntries);
        if(mailEntries.length>0){this.entryLength=1}else{this.entryLength=0;}
        this.orderRows = mailEntries.map((email: string) => ({  
          id: this.branchVendorList["id"],       
          vendor_name: this.branchVendorList["vendorKey"],         
          vendor_email: email }
        ));
        this.editData.branchId = Number(this.branchVendorList["vendorBranchID"]);
        this.editData.branchName = this.branchVendorList["branchName"];
        this.searchBranchName = this.branchVendorList["branchName"];
        //console.log(this.searchBranchName)
        this.editData.makercode = Number(this.authenticationService.getUser().pfId);
      },
      (error) => {
        console.error('Error fetching records:', error);
      }
    );
  }

  exportExcel() {
    import("xlsx").then(xlsx => {

        { field: 'branchName', header: 'Branch Name' },

        { field: 'circleUserPFID', header: 'Circle User PFID' },

      //console.log('Selected Indents:', this.selectedcircleUserAndEmailsArray);

      // Use filteredIndents if no selection is made
      const dataToExport = this.selectedcircleUserAndEmailsArray.length ? this.selectedcircleUserAndEmailsArray : this.filteredcircleUserAndEmailsArray;

        let filteredcircleUserAndEmailsArray:any = {};
        this.cols.forEach(col => {              
          //console.log(col.field);
          filteredcircleUserAndEmailsArray[col.header] = indent[col.field];          
        });
        return filteredcircleUserAndEmailsArray;
      });

      // Check the filtered data structure
      //console.log('Filtered Data:', selectedData);

      // Convert the filtered data to a worksheet
      const worksheet = xlsx.utils.json_to_sheet(selectedData);
      const workbook = { Sheets: { 'data': worksheet }, SheetNames: ['data'] };

      // Save the Excel file
      this.saveAsExcelFile(excelBuffer, "Controller and Nodal Officer Mail Details as on ");
    });
  }

  private saveAsExcelFile(buffer: any, fileName: string): void {
    const data: Blob = new Blob([buffer], { type: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet;charset=UTF-8' });
    FileSaver.saveAs(data, fileName + '_' + this.datePipe.transform(new Date(),'dd-MM-yyyy hh:mm aa') + '.xlsx');
  }

                const diffHours = diffMs / (1000 * 60 * 60); // convert ms to hours

                return {
                  ...item,
                  tatHours: diffHours.toFixed(2),  // keep 2 decimal places
                  TAT : diffDays.toFixed(0), // keep 2 decimal places

                return {

  exportToExcelBranchReportGMUK() {
    import("xlsx").then(xlsx => {
      // Check if selectedIndents is populated
      this.cols = [
        { field : 'createdAt', header: 'IndentDate'},
        { field : 'indentID', header: 'IndentID'},
        { field : 'branchName', header: 'Branch'},
        { field : 'circleCode', header: 'Circle'},
        { field : 'currencyType', header: 'Currency'},
        { field : 'requestedCurrencyValue', header: 'Requested Amount'},
        { field : 'receivedCurrencyValue', header: 'Received Amount'},
        { field : 'vendorBranch', header: 'Vendor'},
        { field : 'indentStatus', header: 'Status'},
        { field : 'requestedDate', header: 'RequestedDate'},
        { field : 'receivedDate', header: 'CompletedDate'},
        { field : 'tatHours', header: 'TAT(Hrs)'},
        { field : 'TAT', header: 'TAT(days)'}
      ];
      // Use filteredIndents if no selection is made
      const dataToExport = this.selectedbranchReportDataArray.length ? this.selectedbranchReportDataArray : this.filteredbranchReportData;
      // Filter the data to export
      const selectedData = dataToExport.map((indent:any) => {
        let filteredbranchReportData:any = {};
        this.cols.forEach(col => {     
          if(col.field == 'createdAt' || col.field == 'requestedDate' || col.field == 'receivedDate'){;
            filteredbranchReportData[col.header] = indent[col.field]!=null ? new Date(indent[col.field]).toISOString().substring(0,10): ''            
          }  
          else if(col.field == 'indentStatus'){
            filteredbranchReportData[col.header] = this.appStatus[indent[col.field]];            
          }    
          else if(col.field == 'currencyType'){
            filteredbranchReportData[col.header] = this.currencyData[indent[col.field]]+" ("+indent[col.field]+")";
          }     
          else if(col.field == 'circleCode'){
            filteredbranchReportData[col.header] = this.circleNames[indent[col.field]]+" ("+indent[col.field]+")";
          }
          else{
            filteredbranchReportData[col.header] = indent[col.field];
          }                    
        });
        return filteredbranchReportData;
      });

      // Check the filtered data structure
      //console.log('Filtered Data:', selectedData);

      // Convert the filtered data to a worksheet
      const worksheet = xlsx.utils.json_to_sheet(selectedData);
      const workbook = { Sheets: { 'data': worksheet }, SheetNames: ['data'] };
      const excelBuffer: any = xlsx.write(workbook, { bookType: 'xlsx', type: 'array' });

      // Save the Excel file
      this.saveAsExcelFile(excelBuffer, "Branch Indent Report at GMUK as on ");
    });
  }
