public ResponseEntity<StreamingResponseBody> getLargeResponse() {
    StreamingResponseBody responseBody = outputStream -> {
        String hugeData = "Your massive 64KB+ data string...";
        outputStream.write(hugeData.getBytes(StandardCharsets.UTF_8));
        outputStream.flush();
    };
    return ResponseEntity.ok()
            .header(HttpHeaders.CONTENT_TYPE, MediaType.TEXT_PLAIN_VALUE)
            .body(responseBody);
}
