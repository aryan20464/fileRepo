deploy_job:
  stage: deploy
  image: artifactory.jfrog.oec:443/<your-wls-image-name>:<tag>   # ←←← CHANGE THIS
  variables:
    ORACLE_HOME: "/opt/oracle/weblogic"          # ←←← Adjust according to your image
    WDT_HOME: "/opt/wdt/weblogic-deploy"         # ←←← Adjust if WDT is in different location
  before_script:
    - jf config add --url="$JF_URL" --user="$JF_USER" --access-token="$JF_TOKEN" --interactive=false
    - jf rt ping
    # Download only archive if needed, or assume it's already built
    - jf rt dl "itfo-treasury-tss-idea-custom-maven-libs/weblogic-deploy/**" "./" --flat=false --recursive=true || true
    - chmod +x ./weblogic-deploy/bin/*.sh || true
    - echo "ORACLE_HOME = $ORACLE_HOME"
    - echo "WDT_HOME = $WDT_HOME"
    - ls -ld $ORACLE_HOME $WDT_HOME || echo "Warning: Paths may need adjustment"
  script:
    - |
      # === Create minimal model.yaml ===
      cat > model.yaml << EOF
      appDeployments:
        Application:
          IDEAAPP:
            SourcePath: 'wlsdeploy/applications/ideaapp.war'
            Target: 'IDEAAPP'
            ModuleType: war
      EOF

    - |
      # === Deploy using WDT (Online Mode) ===
      ./weblogic-deploy/bin/deployApps.sh \
        -oracle_home $ORACLE_HOME \
        -wlst_path $ORACLE_HOME/wlserver/common/bin \
        -model_file model.yaml \
        -archive_file ./archive.zip \
        -admin_url "$WLS_UAT_ADMIN_URL" \
        -admin_user "$WLS_UAT_USER" \
        -admin_pass_env WLS_UAT_PASSWORD \
        -domain_type WLS \
        -remote \
        -verbose

  only:
    - rfm-integration-devsecops
