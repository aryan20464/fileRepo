:: 1. Set your configuration variables
set "GITLAB_URL=https://example.com"
set "TOKEN=your_personal_access_token"

:: 2. Fetch projects, loop through branches, and delete protection
for /f "tokens=1,2" %A in ('curl -s --header "PRIVATE-TOKEN: %TOKEN%" "%GITLAB_URL%/api/v4/projects?per_page=100" ^| jq -r ".[] | \"\(.id) \(.path_with_namespace)\""') do (
    echo Checking Project: %B
    
    for /f "delims=" %C in ('curl -s --header "PRIVATE-TOKEN: %TOKEN%" "%GITLAB_URL%/api/v4/projects/%A/protected_branches" ^| jq -r ".[].name"') do (
        echo --> Unprotecting: %C
        
        :: Note: This simple loop assumes standard branch names. If names have slashes like "feature/test", use the PowerShell method above.
        curl -s -X DELETE --header "PRIVATE-TOKEN: %TOKEN%" "%GITLAB_URL%/api/v4/projects/%A/protected_branches/%C"
    )
)
