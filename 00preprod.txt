exportExcel() {
  import("xlsx").then(xlsx => {
    const dataToExport = this.selectedIndents.length ? this.selectedIndents : this.filteredIndents;

    const keyMap: { [key: string]: string } = {
      'indentID': 'Indent ID',
      'branchCode': 'Branch Code',
      'branchName': 'Branch Name',
      'circleCode': 'Circle Code',
      'vendorRes': 'Vendor/Branch',
      'currencyType': 'Currency Type',
      'currencyMode': 'Currency Mode',
      'requestedCurrencyValue': 'Requested Value',
      'receivedCurrencyValue': 'Received Value',
      'requestedDate': 'Requested Date',
      'receivedDate': 'Received Date',
      'indentStatus': 'Status'
    };

    const selectedData = dataToExport.map(indent => {
      let renamedIndent: any = {};
      
      Object.keys(indent).forEach(key => {
        const friendlyKey = keyMap[key] || key;
        let value = indent[key];

        // Format code if you still need specific conversions from your previous implementation
        switch (key) {
          case 'vendorRes':
            // If data is already parsed as shown in your JSON sample, keep it directly
            value = indent[key] ? indent[key] : '';
            break;
          case 'requestedDate':
          case 'receivedDate':
            value = indent[key] ? indent[key] : 'N/A'; // Keeps formatting or sets N/A for null values
            break;
          default:
            value = indent[key] !== null ? indent[key] : '';
        }

        renamedIndent[friendlyKey] = value;
      });
      return renamedIndent;
    });

    // Generate Excel sheet with new human-readable column headers
    const worksheet = xlsx.utils.json_to_sheet(selectedData);
    const workbook = { Sheets: { 'data': worksheet }, SheetNames: ['data'] };
    const excelBuffer: any = xlsx.write(workbook, { bookType: 'xlsx', type: 'array' });

    this.saveAsExcelFile(excelBuffer, "Indents of " + this.linkedBcode + "-" + this.branchName + " as on ");
  });
}
