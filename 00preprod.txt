@echo off

set GITLAB_URL=https://gitlab.company.com
set TOKEN=YOUR_ADMIN_TOKEN

del gitlab_repos.txt 2>nul

for /L %%p in (1,1,50) do (

    echo Fetching page %%p

    curl --silent ^
    --header "PRIVATE-TOKEN: %TOKEN%" ^
    "%GITLAB_URL%/api/v4/projects?owned=false^&membership=false^&simple=true^&per_page=100^&page=%%p" ^
    >> temp.json

)

findstr "ssh_url_to_repo" temp.json > gitlab_repos.txt

del temp.json

echo Done. Output saved to gitlab_repos.txt
