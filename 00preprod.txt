  colorMap: { [key: string]: string } = {};  // Store the color for each indentID
  previousColorIndex: number = -1;  // To track the previous color index

  getRowColor(indentID: string): string {
    // Check if the color for this indentID is already assigned
    if (!this.colorMap[indentID]) {
      // Get a color index ensuring no two successive indentIDs have the same color
      let colorIndex: number;
  
      do {
        // Pick a random color index
        //colorIndex = Math.floor(Math.random() * this.colors.length); // removed this as per ISD observations.
        colorIndex = this.randomNumberService.generateRandomNumber(this.colors.length);
        

    } while (colorIndex === this.previousColorIndex); // Ensure no same color for successive indentIDs
 
    // Assign color and update the previous color index
    this.colorMap[indentID] = this.colors[colorIndex];
    this.previousColorIndex = colorIndex; // Update previous color index
  }
 
    return this.colorMap[indentID];  // Return the assigned color
  }
    
  
  getUniqueValues(){
    this.uniqueIndents = Array.from(new Set(this.records.map((record:any) => record['indentID'])));
  }


  getUniqueIndentCounts(records: any[]): { [key: string]: number } {
    const indentCountsMap = new Map(); 
   
    records.forEach(record => {
      const indentId = record.indentID;
      indentCountsMap.set(indentId, (indentCountsMap.get(indentId) || 0) + 1);
    });
   
    // Manually construct the object
    const result: { [key: string]: number } = {};
    indentCountsMap.forEach((count, indent_id) => {
      result[indent_id] = count;
    });
    return result;
  }


  getUniqueIndentRecords(records: any[]): { indent_number: any, request_date: any, status: any }[] {
    const uniqueRecordsMap = new Map();
   
    records.forEach(record => {
      if (!uniqueRecordsMap.has(record.indentID)) {
        uniqueRecordsMap.set(record.indentID, {
          indent_number: record.indentID,
          request_date: record.createdAt,
          status: record.indentStatus
        });
      }
    });
   
    //console.log(Array.from(uniqueRecordsMap.values()));
    return Array.from(uniqueRecordsMap.values());
  }

  customFilter(value: string, field: string, mode: string): void {
    const filteredValue = value.toLowerCase();
    this.filteredIndents = this.allIndents.filter((sdata:any) => {
      switch (field) {
        case 'vendorRes':
          return (this.branchVendorList[sdata.vendorRes]+sdata.vendorBranch).toLowerCase().includes(filteredValue);
        case 'currencyType':
          return this.currencyData[sdata.currencyType].toLowerCase().includes(filteredValue);
        case 'currencyMode':
          return this.currencyMode[sdata.currencyMode].toLowerCase().includes(filteredValue);
        case 'indentStatus':
          return this.appStatus[sdata.indentStatus].toLowerCase().includes(filteredValue);
        default:
          return sdata[field].toString().toLowerCase().includes(filteredValue);
      }
    });
  }

  exportExcel() {
    import("xlsx").then(xlsx => {
      // Check if selectedIndents is populated
      //console.log('Selected Indents:', this.selectedIndents);

      // Use filteredIndents if no selection is made
      const dataToExport = this.selectedIndents.length ? this.selectedIndents : this.filteredIndents;

      // Filter the data to export
      const selectedData = dataToExport.map(indent => {
        let filteredIndent:any = {};
        this.cols.forEach(col => {              
          //console.log(col.field);
          filteredIndent[col.field] = indent[col.field];
          switch (col.field) {
            case 'vendorRes':
              filteredIndent[col.field] = this.branchVendorList[indent[col.field]]+" - "+indent["vendorBranch"];break;
            case 'currencyType':
              filteredIndent[col.field] = this.currencyData[indent[col.field]]+" ("+indent[col.field]+")";break;
            case 'currencyMode':
              filteredIndent[col.field] = this.currencyMode[indent[col.field]];break;
            case 'indentStatus':
              filteredIndent[col.field] = this.appStatus[indent[col.field]];break;
            case 'requestedDate':
              filteredIndent[col.field] = this.convertHRF(indent[col.field]);break;
            case 'receivedDate':
              filteredIndent[col.field] = this.convertHRF(indent[col.field]);break; 
            case 'circleCode':
              filteredIndent[col.field] = this.circleNames[indent[col.field]];break; 
            default:
              filteredIndent[col.field] = indent[col.field];
          }
        });
        return filteredIndent;
      });

      // Check the filtered data structure
      //console.log('Filtered Data:', selectedData);

      // Convert the filtered data to a worksheet
      const worksheet = xlsx.utils.json_to_sheet(selectedData);
      const workbook = { Sheets: { 'data': worksheet }, SheetNames: ['data'] };
      const excelBuffer: any = xlsx.write(workbook, { bookType: 'xlsx', type: 'array' });

      // Save the Excel file
      this.saveAsExcelFile(excelBuffer, "Indents of "+this.linkedBcode+"-"+this.branchName+" as on ");
    });
  }

  private saveAsExcelFile(buffer: any, fileName: string): void {
    const data: Blob = new Blob([buffer], { type: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet;charset=UTF-8' });    
    FileSaver.saveAs(data, fileName + this.datePipe.transform(new Date(),'dd-MM-yyyy hh:mm aa') + '.xlsx');
  }
