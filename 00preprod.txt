package in.co.sbi.rfx.indents.dto;

import in.co.sbi.rfx.indents.validation.EmailIdentifier;
import in.co.sbi.rfx.indents.validation.NumericCode;
import in.co.sbi.rfx.indents.validation.SafeIdentifier;
import jakarta.validation.Valid;
import lombok.Getter;
import lombok.Setter;

@Getter
@Setter
@Valid
public class IndentsDTO {
    @NumericCode
    private int currency;
    @SafeIdentifier
    private String currency_mode;
    @SafeIdentifier
    private String vendor_name;
    @SafeIdentifier
    private String exchange;
    @EmailIdentifier
    private String vendor_email;
    @SafeIdentifier
    private String vendor_branch;

    // Constructors, Getters, and Setters
    public IndentsDTO() {}
}

package in.co.sbi.rfx.indents.dto;

import com.fasterxml.jackson.annotation.JsonIgnore;
import in.co.sbi.rfx.indents.validation.EmailIdentifier;
import in.co.sbi.rfx.indents.validation.NumericCode;
import in.co.sbi.rfx.indents.validation.SafeIdentifier;
import jakarta.validation.Valid;
import jakarta.validation.constraints.Email;
import lombok.Getter;
import lombok.Setter;

import java.util.List;

@Getter
@Setter
public class IndentsRequestDTO {
    @NumericCode
    private String bcode;
    @EmailIdentifier
    private String branchmail;
    @SafeIdentifier
    private String branchname;
    private boolean declaration_check;
    @SafeIdentifier
    private String request_date;
    @JsonIgnore
    private int makercode;
    @SafeIdentifier
    private String makerName;
    @NumericCode
    private int status;
    @Valid
    private List<IndentsDTO> entries;
}
