- >
      ENDPOINTS=(
        "https://example.com"
        "https://example.com"
        "http://service.local"
      );
      EXIT_CODE=0;
      for url in "${ENDPOINTS[@]}"; do
        echo "Checking $url ...";
        if curl -f --max-time 5 -s "$url" > /dev/null; then
          echo "  -> Success";
        else
          echo "  -> Connection to $url failed!";
          EXIT_CODE=1;
        fi;
      done;
      exit $EXIT_CODE;
