image: artifactory.jfrog.sbi:443/dso-base-image/java-maven-images/rhel-jdk-17-maven3.8.5-custom:1.0.2

variables:
  MAVEN_CLI_OPTS: "-s .m2/settings.xml --batch-mode"

build_job:
  stage: build
  script:
    - mvn clean package -U -P uat
  artifacts:
    paths:
      - target/
    expire_in: 1 day

deploy_job:
  stage: deploy
  before_script:
    - jf config add --url="$JF_URL" --user="$JF_USER" --access-token="$JF_TOKEN" --interactive=false
    - jf config use itfo-server
    - jf rt ping

    - jf rt dl "itfo-treasury-tss-idea-custom-maven-libs/weblogic-deploy/**" "./weblogic-deploy/" --flat=false --recursive=true
    - chmod +x weblogic-deploy/bin/*.sh
    - cp target/RfmIndentsService.war .
  script:
    - |
      mkdir -p wlsdeploy/applications
      cp RfmIndentsService.war wlsdeploy/applications/

      cd wlsdeploy
      zip -r ../archive.zip .
      cd ..

      ./weblogic-deploy/bin/deployApps.sh \
        -admin_url "$WLS_UAT_ADMIN_URL" \
        -user "$WLS_UAT_USER" \
        -password "$WLS_UAT_PASSWORD" \
        -archive_file ./archive.zip \
        -target "IDEAAPP" \
        -verbose
  only:
    - rfm-integration-devsecops
