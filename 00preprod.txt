http {

    default_type  application/octet-stream;

    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';

    access_log  /var/log/nginx/access.log  main;
    error_log   /var/log/nginx/error.log   debug;

    sendfile        on;
    #tcp_nopush     on;

    #keepalive_timeout  0;
    keepalive_timeout  65;

    #gzip  on;


    server {
        listen       4200;
        listen  [::]:4200;
        server_name  localhost;
        server_tokens off;

       location / {
        root   /usr/share/nginx/html;
        index  index.html index.htm;
        }

        error_page   500 502 503 504  /50x.html;
        location = /50x.html {
            root   /usr/share/nginx/html;
        }

         location /rfx/ {
             alias /usr/share/nginx/html/rfx/browser/;
             try_files $uri$args $uri$args/ /rfx/index.html;
        }

            proxy_set_header        X-Real-IP       $remote_addr;
            proxy_pass http://rfx-app-manager-service:9190/rfx/rfxmanagerservice;

        location /middlewarerfxAuthService{
            proxy_set_header       X-Real-IP       $remote_addr;
            proxy_pass http://rfx-auth-service:5202/rfx/rfxauthservice;
        }

        location /middlewarerfxCryptoService{
            proxy_set_header       X-Real-IP       $remote_addr;
            proxy_pass http://rfx-crypto-service:8085/rfx/rfxcryptoservice;
        }



        location /grafana/ {
           proxy_pass http://grafana:3000;
           proxy_set_header   Host $host;
           proxy_set_header   X-Real-IP $remote_addr;
           proxy_set_header   X-Forwarded-For $proxy_add_x_forwarded_for;
           proxy_set_header   X-Forwarded-Proto $scheme;
           #rewrite ^/grafana/(.*) /$1 break;
        }


    }
