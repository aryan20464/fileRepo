apiVersion: apps/v1
kind: Deployment
metadata:
  name: rforex-fcra
  namespace: forex
  labels:
    app: rforex-fcra
spec:
  replicas: 1
  selector:
    matchLabels:
      app: rforex-fcra
  template:
    metadata:
      labels:
        app: rforex-fcra
    spec:
      securityContext:
        runAsNonRoot: true
        runAsUser: 1001
        fsGroup: 1001               # ← critical: allows user 1001 to write to mounted volume

      containers:
        - name: rforex-fcra
          image: scm/rforex-fcra:prp-1.0
          imagePullPolicy: Always
          ports:
            - containerPort: 8081

          # Recommended: adjust paths according to your actual health endpoints
          livenessProbe:
            httpGet:
              path: /actuator/health/liveness   # or /health, /api/health, etc.
              port: 8081
            initialDelaySeconds: 60
            periodSeconds: 10
            timeoutSeconds: 3
            failureThreshold: 5

          readinessProbe:
            httpGet:
              path: /actuator/health/readiness  # or same as liveness if you don't distinguish
              port: 8081
            initialDelaySeconds: 30
            periodSeconds: 5
            timeoutSeconds: 3
            failureThreshold: 3

          volumeMounts:
            - name: fcra-storage
              mountPath: /fcra

          securityContext:
            allowPrivilegeEscalation: false
            capabilities:
              drop:
                - ALL
            seccompProfile:
              type: RuntimeDefault

      volumes:
        - name: fcra-storage
          persistentVolumeClaim:
            claimName: pvc-fcra

---
apiVersion: v1
kind: Service
metadata:
  name: rforex-fcra-service
  namespace: forex
spec:
  selector:
    app: rforex-fcra
  type: ClusterIP
  ports:
    - port: 8081
      targetPort: 8081
      protocol: TCP
