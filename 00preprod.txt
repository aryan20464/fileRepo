stages:
  - build
  - deploy

variables:
  MAVEN_REPO: "${CI_PROJECT_DIR}/.m2/repository"   # Better to use project dir for caching
  MAVEN_CLI_OPTS: "-B --no-transfer-progress -Dmaven.test.skip=true -Dmaven.repo.local=${MAVEN_REPO}"
  MAVEN_HOME: "/devtools/maven/apache-maven-3.8.5"
  JAVA_HOME: "/devtools/jdk21/jdk-21.0.6"

cache:                                      # ← Add this at root level
  key: ${CI_COMMIT_REF_SLUG}-maven
  paths:
    - .m2/repository/

build-war:
  stage: build
  tags:
    - cicd-fileupload-auto
  script:
    - echo "Using Maven repo at $MAVEN_REPO"
    - mkdir -p $MAVEN_REPO
    - export PATH="$MAVEN_HOME/bin:$PATH"
    
    - java -version
    - mvn -version
    
    # First run: download dependencies (no -o)
    - mvn $MAVEN_CLI_OPTS dependency:go-offline -U -P preprod -s /devtools/gitlab-runner/.m2/settings.xml || true
    
    # Actual build (still without -o)
    - mvn $MAVEN_CLI_OPTS clean package -U -X -P preprod -s /devtools/gitlab-runner/.m2/settings.xml
  artifacts:
    paths:
      - target/*.war
  only:
    - rforex-integration
