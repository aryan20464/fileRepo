@Repository
public interface IndentsBranchDetailsRepository extends JpaRepository<BranchIndentBranchList, Long>
{
    @Query("SELECT bi.emailId FROM BranchIndentBranchList bi where bi.branchCode = :bcode")
    String getBranchDetails(@Pattern(regexp = "^\\d+$", message = "bcode must contain numbers only") @Param("bcode") String bcode,@Pattern(regexp = "^\\d+$", message = "user must contain numbers only") @Param("user") String user);


}

@Valid
    public String getBranchEmail(@Pattern(regexp = "^\\d+$", message = "bcode must contain numbers only") String bcode, @Pattern(regexp = "^\\d+$", message = "user must contain numbers only") String user) throws NullPointerException
    {
        try {
            if(bcode.matches("^[0-9]+$") && user.matches("^[0-9]+$")){
            String branchEmail = indentsBranchDetailsRepository.getBranchDetails(bcode,user);
            return org.owasp.encoder.Encode.forHtml(branchEmail);}
            else{
                return org.owasp.encoder.Encode.forHtml("");
            }
        }

@GetMapping("/getBranchEmail")
    @Valid
    public String getBranchEmail(@Pattern(regexp = "^\\d+$", message = "bcode must contain numbers only") @RequestParam("bcode") String bcode) throws NullPointerException
    {
        String user = currentUser.getPfId();
        String userBCode = currentUser.getBranchCode();
        if(userBCode.matches("^[0-9]+$") && user.matches("^[0-9]+$")){
        return indentService.getBranchEmail(userBCode,user);}
        else{
            return "";
        }

    }
