http {
    ...

    map $http_x_forwarded_proto $real_proto {
        default $http_x_forwarded_proto;
        ''      $scheme;
    }

    server {
        listen 4200;
        listen [::]:4200;
        server_name _;                 # ← changed

        location / {
            root /usr/share/nginx/html;
            index index.html index.htm;
            try_files $uri $uri/ /index.html;   # usually better for SPA
        }

        location /rfx/ {
            alias /usr/share/nginx/html/rfx/browser/;
            try_files $uri $uri/ /rfx/index.html;
        }

        location /middlewarerfxManagerService {
            proxy_set_header Host               $http_host;
            proxy_set_header X-Real-IP          $remote_addr;
            proxy_set_header X-Forwarded-For    $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto  $real_proto;
            proxy_set_header X-Forwarded-Host   $http_x_forwarded_host;
            proxy_set_header X-Forwarded-Port   $http_x_forwarded_port;
            proxy_pass http://rfx-app-manager-service:9190/rfx/rfxmanagerservice;
        }

        location /middlewarerfxAuthService {
            # same headers as above
            proxy_set_header Host               $http_host;
            proxy_set_header X-Real-IP          $remote_addr;
            proxy_set_header X-Forwarded-For    $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto  $real_proto;
            proxy_set_header X-Forwarded-Host   $http_x_forwarded_host;
            proxy_set_header X-Forwarded-Port   $http_x_forwarded_port;
            proxy_pass http://rfx-auth-service:5202/rfx/rfxauthservice;
        }

        # same for CryptoService and Grafana

        location /grafana/ {
            # keep your existing + add the missing
            proxy_set_header Host               $http_host;
            proxy_set_header X-Real-IP          $remote_addr;
            proxy_set_header X-Forwarded-For    $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto  $real_proto;
            proxy_set_header X-Forwarded-Host   $http_x_forwarded_host;
            proxy_set_header X-Forwarded-Port   $http_x_forwarded_port;
            proxy_pass http://grafana:3000/;
        }
    }
}
