package com.example.security;

public class UserContext {

    private final String pfId;
    private final String branchCode;

    public UserContext(String pfId, String branchCode) {
        this.pfId = pfId;
        this.branchCode = branchCode;
    }

    public String getPfId() {
        return pfId;
    }

    public String getBranchCode() {
        return branchCode;
    }
}






package com.example.security;

public final class UserContextHolder {

    private static final ThreadLocal<UserContext> USER_CONTEXT =
            new ThreadLocal<>();

    private UserContextHolder() {
    }

    public static void set(UserContext context) {
        USER_CONTEXT.set(context);
    }

    public static UserContext get() {
        return USER_CONTEXT.get();
    }

    public static void clear() {
        USER_CONTEXT.remove();
    }
}





private void extractUserInfo(String payloadJson) {

    try {

        ObjectMapper mapper = new ObjectMapper();
        JsonNode root = mapper.readTree(payloadJson);

        JsonNode user = root.get("user");

        if (user != null) {

            String pfId =
                    user.path("pfId").asText("unknown");

            String branch =
                    user.path("branchCode").asText("unknown");

            MDC.put(
                    "user_info",
                    String.format("PF-%s|BCODE-%s", pfId, branch)
            );

            UserContextHolder.set(
                    new UserContext(pfId, branch)
            );
        }

    } catch (Exception e) {

        MDC.put("user_info", "token-un-parsable");

        UserContextHolder.set(
                new UserContext("unknown", "unknown")
        );
    }
}

finally {

    MDC.remove("clientIp");
    MDC.remove("user_info");

    UserContextHolder.clear();
}



package com.example.security;

import org.springframework.stereotype.Component;

@Component
public class CurrentUser {

    public String getPfId() {

        UserContext context =
                UserContextHolder.get();

        return context != null
                ? context.getPfId()
                : "unknown";
    }

    public String getBranchCode() {

        UserContext context =
                UserContextHolder.get();

        return context != null
                ? context.getBranchCode()
                : "unknown";
    }

    public UserContext getUserContext() {
        return UserContextHolder.get();
    }
}
