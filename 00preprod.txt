onDateSubmit(form: any) {
    if (form.valid ) {
      //console.log("from date "+this.fromDate);
      //console.log("to date "+this.toDate);
      const payload = {
        fromDate: this.fromDate,
        toDate: this.toDate,
      }
      this.http.post(`${environment.rfmIndentService}/showIndentReportAtGMUK`, payload)
      .subscribe({
        next: (response: any[]) => {
          this.branchReportData = response || [];
          this.branchReportData = this.branchReportData.map(item => {
              const requestedDate = new Date(item.requestedDate);
              const receivedDate = item.receivedDate ? new Date(item.receivedDate) : null;

              if (receivedDate) {
                const diffMs = receivedDate.getTime() - requestedDate.getTime(); // milliseconds
                const diffHours = diffMs / (1000 * 60 * 60); // convert ms to hours
                const diffDays = diffMs / (1000 * 60 * 60 * 24); // convert hrs to days

                return {
                  ...item,
                  tatHours: diffHours.toFixed(2),  // keep 2 decimal places
                  TAT : diffDays.toFixed(0), // keep 2 decimal places
                };
              } else {
                return {
                  ...item,
                  tatHours: '',  // or null, or 'Pending'
                  TAT : '', // or null, or 'Pending'
                };
              }
            });
          this.filteredbranchReportData = [...this.branchReportData];
          //console.log(this.branchReportData);
          this.tableHeaders = this.branchReportData.length > 0 ? Object.keys(this.branchReportData[0]) : [];
          this.submittedReports = true;
        },
        error: (error) => {
          //console.error('Error fetching data', error);
          this.branchReportData = [];
          this.tableHeaders = [];
          this.submittedReports = true;
        }
      });  }
  }
