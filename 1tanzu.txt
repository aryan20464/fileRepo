apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: minimal-ingress
  namespace: uatcmm
  annotations:
    nginx.ingress.kubernetes.io/rewrite-target: /
spec:
  ingressClassName: uatcmm
  rules:
  - host: ""   # Accept requests to IP directly
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: nginx-cmm-service
            port:
              number: 80
